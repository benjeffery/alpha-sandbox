 17 	/**
 18 	 * A condition bar represents a condition that can be configured by the user via
 19 	 * UI. Components (select and input boxes) can be retrived by the interface and
 20 	 * events are automatically managed by the callback functions provided.
 21 	 *
 22 	 * @param id {Integer} An integer that identifies uniquely the condition bar.
 23 	 * @param propertyMap {Object} A hash that associates property names to
 24 	 * properties.
 25 	 * @param operatorSymbolMap {Object} A hash that associates operator symbols to
 26 	 * operators.
 27 	 * @param valuesUpdateFunction {Function} The function that will be called when
 28 	 * the input values are updated.
 29 	 * @param cqlUpdateFunction {Function} The function in charge of updating the
 30 	 * generated cql code when this condition changes.
 31 	 * @param tooltipDivName {String} The base name of the container to be updated
 32 	 * with the description of the selected property.
 33 	 *
 34 	 * @name expcat.plugins.ConditionBar
 35 	 * @constructor
 36 	 * @public
 37 	 * @author Jacob Almagro - ExplorerCat Project
 38 	 */
 39 
 40 	var ConditionBar = function(id, propertyMap, operatorSymbolMap, valuesUpdateFunction, cqlUpdateFunction, tooltipDivName) {
 41 
 42 		var conditionId = id;
 43 		var propertyDictionary = propertyMap;
 44 		var operatorsBySymbol = operatorSymbolMap;
 45 		var valuesUpdateCallback = valuesUpdateFunction;
 46 		var cqlUpdateCallback = cqlUpdateFunction;
 47 		var isNegated = false;
 48 		var tooltipBaseName = tooltipDivName || "helpTooltip";
 49 
 50 		// Select boxes and inputs
 51 		var selectProperty;
 52 		var selectOperator;
 53 		var inputValues = [];
 54 
 55 		/**
 56 		 * Initializes the object.
 57 		 * @memberOf expcat.plugins.ConditionBar#
 58 		 * @private
 59 		 * @ignore
 60 		 */
 61 
 62 		var init = function() {
 63 			selectProperty = createSelectBox("selectProperty" + id, getMapKeys(propertyDictionary));
 64 			selectOperator = createSelectBox("selectOperator" + id, getMapKeys(operatorsBySymbol));
 65 			updateValues();
 66 
 67 			// Handlers for the property box.
 68 			selectProperty.bind("change", function() {
 69 				updateOperators();
 70 				updateValues();
 71 				valuesUpdateCallback(conditionId, inputValues);
 72 				resetInvalidValues();
 73 				updateTooltipText();
 74 				cqlUpdateCallback();
 75 			});
 76 
 77 			// Handlers for the operator box.
 78 			selectOperator.bind("change", function() {
 79 				updateValues();
 80 				valuesUpdateCallback(conditionId, inputValues);
 81 				resetInvalidValues();
 82 				cqlUpdateCallback();
 83 			});
 84 
 85 		};
 86 
 87 		/**
 88 		 * Creates a new select box with the given name and values (options).
 89 		 * @param name {String} The name of the select box.
 90 		 * @param values {Array} Array of text values for the select box.
 91 		 * @return {jQuery} The configured select box.
 92 		 *
 93 		 * @memberOf expcat.plugins.ConditionBar#
 94 		 * @private
 95 		 * @ignore
 96 		 */
 97 
 98 		var createSelectBox = function(name, values) {
 99 			var selectBox = $("<select></select>");
100 			var i;
101 
102 			selectBox.attr("name", name);
103 			selectBox.attr("id", name);
104 
105 			if(values && values.length > 0) {
106 				setSelectOptions(selectBox, values);
107 			}
108 
109 			return selectBox;
110 		}
111 
112 		/**
113 		 * Sets the options of the given selection box using the values parameter.
114 		 * @param selectBox {jQuery} The select object whose options will be set.
115 		 * @param valeus {Array} An array of text values that will be used to create the
116 		 * options.
117 		 *
118 		 * @memberOf expcat.plugins.ConditionBar#
119 		 * @private
120 		 * @ignore
121 		 */
122 
123 		var setSelectOptions = function(selectBox, values) {
124 			var optionsHTML = [];
125 			var currentOption;
126 
127 			// Clear the options.
128 			selectBox.html("");
129 
130 			for( i = 0; i < values.length; ++i) {
131 				currentOption = "<option value=\"" + i + "\">" + values[i] + "</option>";
132 				optionsHTML.push(currentOption);
133 			}
134 
135 			// Add the options
136 			selectBox.html(optionsHTML.join(" "));
137 			selectBox.val(0);
138 		}
139 
140 		/**
141 		 * Updates the operators that are shown based on the selected
142 		 * property.
143 		 *
144 		 * @memberOf expcat.plugins.ConditionBar#
145 		 * @private
146 		 * @ignore
147 		 */
148 
149 		var updateOperators = function() {
150 			var supportedOperators = getOperatorsSupportedBySelectedProperty();
151 			var operatorSymbols = [];
152 			var i;
153 
154 			for( i = 0; i < supportedOperators.length; ++i) {
155 				operatorSymbols.push(supportedOperators[i].getSymbol());
156 			}
157 
158 			setSelectOptions(selectOperator, operatorSymbols);
159 		}
160 
161 		/**
162 		 * Gets the operators that are supported by the selected property.
163 		 * @return {Array} An array containing the supported operators.
164 		 *
165 		 * @memberOf expcat.plugins.ConditionBar#
166 		 * @private
167 		 * @ignore
168 		 */
169 
170 		var getOperatorsSupportedBySelectedProperty = function() {
171 			var selectedProperty = getSelectedProperty();
172 			var supportedOperators = [];
173 			var op;
174 
175 			for(op in operatorsBySymbol) {
176 				if(operatorsBySymbol[op].supportsType(selectedProperty.getType()))
177 					supportedOperators.push(operatorsBySymbol[op]);
178 			}
179 
180 			return supportedOperators;
181 		}
182 
183 		/**
184 		 * Updates the input values according to the selected
185 		 * operator.
186 		 *
187 		 * @memberOf expcat.plugins.ConditionBar#
188 		 * @private
189 		 * @ignore
190 		 */
191 
192 		var updateValues = function() {
193 			var operator = getSelectedOperator();
194 			if(inputValues.length !== operator.getArity() - 1) {
195 				inputValues = createInputValues(operator.getArity() - 1);
196 			}
197 		}
198 
199 		/**
200 		 * Resets any invalid input value to a default.
201 		 *
202 		 * @memberOf expcat.plugins.ConditionBar#
203 		 * @private
204 		 * @ignore
205 		 */
206 
207 		var resetInvalidValues = function() {
208 			var i;
209 			for( i = 0; i < inputValues.length; ++i) {
210 				resetInvalidValue(inputValues[i]);
211 			}
212 		}
213 
214 		/**
215 		 * Resets the given input to a default if its value is invalid.
216 
217 		 * @memberOf expcat.plugins.ConditionBar#
218 		 * @private
219 		 * @ignore
220 		 */
221 
222 		var resetInvalidValue = function(inputValue) {
223 			if(!isValueValid(inputValue)) {
224 				setDefaultValue(inputValue);
225 			}
226 		}
227 
228 		/**
229 		 * Sets the given input with the default value of the property currently
230 		 * selected.
231 
232 		 * @memberOf expcat.plugins.ConditionBar#
233 		 * @private
234 		 * @ignore
235 		 */
236 
237 		var setDefaultValue = function(input) {
238 			var property = getSelectedProperty();
239 			var minimum = property.getMinimum();
240 			var maximum = property.getMaximum();
241 			var allowedValues = property.getAllowedValues();
242 			var type = property.getType();
243 
244 			if(allowedValues !== null) {
245 				input.val(allowedValues[0]);
246 			}
247 			else if(minimum !== null) {
248 				input.val(minimum);
249 			}
250 			else if(maximum !== null) {
251 				input.val(maximum);
252 			}
253 			else {
254 				input.val(type.defaultValue());
255 			}
256 		}

258 		/**
259 		 * Creates an array of input boxes (or select boxes) to store the condition
260 		 * values.
261 		 * @param numValues The number of text inputs to be created.
262 		 * @return {Array} An array of text input boxes (jQuery objects0) for the values.
263 		 *
264 		 * @memberOf expcat.plugins.ConditionBar#
265 		 * @private
266 		 * @ignore
267 		 */
268 
269 		var createInputValues = function(numValues) {
270 			var inputBoxes = [];
271 			var currentInput;
272 			var inputId;
273 			var width = 80 / numValues;
274 			var i;
275 
276 			for( i = 0; i < numValues; ++i) {
277 				currentInput = $("<input type='text'></input>");
278 				inputId = "inputValue_" + i + "_" + conditionId
279 				currentInput.attr("name", inputId);
280 				currentInput.attr("id", inputId);
281 				currentInput.css("width", width + "%");
282 
283 				inputBoxes.push(currentInput);
284 
285 				// Hanlders for the input.
286 				currentInput.bind("change", function() {
287 					resetInvalidValue($(this));
288 					cqlUpdateCallback();
289 				});
290 
291 			}
292 
293 			return inputBoxes;
294 		}

322 		}

324 		/**
325 		 * Checks the value provided are coherent with the property and operator
326 		 * selected. In case the value is valid it will be reparsed to trim any added
327 		 * rubbish. References to other properties (as strings) are considered valid
328 		 * without type checking.
329 		 * @return {Boolean} True if the value is valid (coherent with the selected
330 		 * property).
331 		 *
332 		 * @memberOf expcat.plugins.ConditionBar#
333 		 * @public
334 		 */
335 
336 		var isValueValid = function(inputValue) {
337 			var property = getSelectedProperty();
338 			var isValid = property.isValueValid(new Value(property.getType().getName(), inputValue.val()));
339 			var i;
340 
341 			/** Check if we are referencing another property */
342 			if(!isValid && propertyDictionary[inputValue.val()]) {
343 				return true;
344 			}
345 
346 			/** Re-parse the input value to remove any rubbish **/
347 			if(isValid) {
348 				inputValue.val(property.getType().parse(inputValue.val()));
349 			}
350 
351 			return isValid;
352 		}

440 		/**
441 		 * Builds a clause for the condition.
442 		 * @retun A clause object that encapsulates the condition and can be directly
443 		 * translated into CQL.
444 		 *
445 		 * @memberOf expcat.plugins.ConditionBar#
446 		 * @public
447 		 */
448 
449 		var buildClause = function() {
450 			var clause = new Clause();
451 			var property = getSelectedProperty();
452 			var clauseValues = [property];
453 			var i;
454 
455 			clause.setOperator(getSelectedOperator().getName());
456 			for( i = 0; i < inputValues.length; ++i) {
457 				clauseValues.push(new Value(property.getType().getName(), inputValues[i].val()));
458 			}
459 
460 			clause.setOperands(clauseValues);
461 
462 			if(isConditionNegated()) {
463 				var negatedClause = new Clause();
464 				negatedClause.setOperator("NOT");
465 				negatedClause.setOperands([clause]);
466 				return negatedClause;
467 			}
468 
469 			return clause;
470 		}

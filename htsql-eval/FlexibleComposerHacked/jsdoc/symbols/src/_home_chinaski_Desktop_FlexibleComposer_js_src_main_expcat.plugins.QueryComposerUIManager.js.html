 12 expcat.plugins.QueryComposerUIManager = ( function() {

 14 	// Aliases
 15 	var OPERATORS = expcat.cql.Operator;
 16 	var Property = expcat.cql.Property;
 17 	var ConditionBar = expcat.plugins.ConditionBar;
 18 	var OperatorBar = expcat.plugins.OperatorBar;
 19 	var QueryBuilder = expcat.cql.QueryBuilder;

 21 	/**
 22 	 * Manages the UI for the query composer component. This object is in charge of
 23 	 * directing the UI creation and sync.
 24 	 *
 25 	 * The constructor creates a new manager.
 26 	 *
 27 	 * @param operators {[expcat.cql.Operator]}
 28 	 * @param propertyDictionary {Object}
 29 	 *
 30 	 * @name expcat.plugins.QueryComposerUIManager
 31 	 * @constructor
 32 	 * @author Jacob Almagro - ExplorerCat Project.
 33 	 */
 34 
 35 	var QueryComposerUIManager = function(ops, properties, nestedDivs, containerId) {
 36 
 37 		var operatorMap = ops;
 38 		var propertyMap = properties;
 39 		var currentId = 0;
 40 		var uiContainerId = containerId;
 41 		var queryElements = [];
 42 		var ui = new expcat.plugins.QueryComposerUI(nestedDivs);
 43 
 44 		var logicalOperatorMap = {
 45 			"AND" : OPERATORS.AND,
 46 			"OR" : OPERATORS.OR
 47 		};
 48 
 49 		/**
 50 		 * Callback function in charge of fixing any incoherent nesting for operators (it
 51 		 * is passed to the QueryComposerUI). Notice that this method modifies the array
 52 		 * given as a parameter.
 53 		 * @param nestingFlags {[Boolean]} The array of flags that defines which
 54 		 * components are nested (true).
 55 
 56 		 * @private
 57 		 * @ignore
 58 		 * @memberOf expcat.plugins.QueryComposerUIManager#
 59 		 */
 60 
 61 		var brokenOpCorrectorFunction = function(nestingFlags) {
 62 			var i;
 63 			for( i = 1; i < nestingFlags.length; i += 2) {
 64 				if(!nestingFlags[i - 1] && nestingFlags[i] || !nestingFlags[i + 1] && nestingFlags[i]) {
 65 					nestingFlags[i] = false;
 66 				}
 67 			}
 68 		};
 69 
 70 		/**
 71 		 * Callback function that is executed when a condition is removed by the user.
 72 		 * @private
 73 		 * @ignore
 74 		 * @memberOf expcat.plugins.QueryComposerUIManager#
 75 		 */
 76 
 77 		var removeConditionHandler = function() {
 78 			var conditionPanelId = "cnd" + $(this).attr("id").substring(3);
 79 			var elementIndex = getIndexForQueryElement(parseInt($(this).attr("id").substring(3)));
 80 
 81 			if(!ui.checkPreviousComponentIsNested(conditionPanelId) && ui.getNumConsecutiveNestedComponents(conditionPanelId) > 1) {
 82 				ui.removeUIComponents(conditionPanelId, 1);
 83 				queryElements.splice(elementIndex, elementIndex < queryElements.length ? 2 : 1);
 84 			}
 85 			else {
 86 				ui.removeUIComponents(conditionPanelId, -1);
 87 				if(elementIndex > 0)
 88 					queryElements.splice(elementIndex - 1, 2);
 89 				else
 90 					queryElements.splice(elementIndex, 1);
 91 			}
 92 
 93 			ui.applyNestingCorrectorFunction(brokenOpCorrectorFunction);
 94 			updateUI();
 95 		};
 96 
 97 		/**
 98 		 * Callback function that is executed when the user clicks on the nest button.
 99 		 * @private
100 		 * @ignore
101 		 * @memberOf expcat.plugins.QueryComposerUIManager#
102 		 */
103 
104 		var nestingHandler = function() {
105 			ui.setComponentNestingWithinRadius("op" + $(this).attr("id").substring(4), 1, true);
106 			updateUI();
107 		};
108 
109 		/**
110 		 * Callback function that is executed when the user clicks on the unnest button.
111 		 * @private
112 		 * @ignore
113 		 * @memberOf expcat.plugins.QueryComposerUIManager#
114 		 */
115 
116 		var unnestingHandler = function() {
117 			ui.setComponentNestingWithinRadius("op" + $(this).attr("id").substring(4), 1, false);
118 			ui.applyNestingCorrectorFunction(brokenOpCorrectorFunction);
119 			updateUI();
120 		};
121 
122 		/**
123 		 * Gets the index fo the given query element.
124 		 * @param elementId The identifier of the element we are looking for.
125 		 *
126 		 * @private
127 		 * @ignore
128 		 * @memberOf expcat.plugins.QueryComposerUIManager#
129 		 */
130 
131 		var getIndexForQueryElement = function(elementId) {
132 			var i;
133 			for( i = 0; i < queryElements.length; ++i) {
134 				if(queryElements[i].getId() === elementId)
135 					return i;
136 			}
137 			return -1;
138 		}
139 
140 		/**
141 		 * Creates an operator panel that represents an OperatorBar instance.
142 		 * @return {jQuery} The DOM element that represents the panel.
143 		 *
144 		 * @private
145 		 * @ignore
146 		 * @memberOf expcat.plugins.QueryComposerUIManager#
147 		 */
148 
149 		var createOperatorPanel = function() {
150 
151 			var operatorBar = new OperatorBar(currentId, logicalOperatorMap, updateCQLCode);
152 			var operatorSelect = operatorBar.getOperatorSelect();
153 			var id = operatorBar.getId();
154 			var operatorPanel, operatorLine, operator, nestButton, unnestButton;
155 
156 			queryElements.push(operatorBar);
157 			operator = createDIV("operator", "operator" + id);
158 			operatorLine = createDIV("operatorLine", "opLine" + id);
159 			operatorPanel = createDIV("operatorPanel", "op" + id);
160 			nestButton = createDIV("nestButton", "nbtn" + id);
161 			unnestButton = createDIV("unnestButton", "ubtn" + id);
162 
163 			nestButton.bind("click", nestingHandler);
164 			unnestButton.bind("click", unnestingHandler);
165 
166 			operatorPanel.append(operatorLine);
167 			operator.append(operatorSelect);
168 			operatorPanel.append(operator);
169 			operatorPanel.append(unnestButton);
170 			operatorPanel.append(nestButton);
171 			currentId++;
172 
173 			return operatorPanel;
174 		};
175 
176 		/**
177 		 * Creates an condition panel that represents a ConditionBar instance.
178 		 * @return {jQuery} The DOM element that represents the panel.
179 		 *
180 		 * @private
181 		 * @ignore
182 		 * @memberOf expcat.plugins.QueryComposerUIManager#
183 		 */
184 
185 		var createConditionPanel = function() {
186 
187 			var conditionBar = new ConditionBar(currentId, propertyMap, operatorMap, updateConditionValues, updateCQLCode);
188 			conditionBar.refresh();
189 
190 			var propertySelect = conditionBar.getPropertySelect();
191 			var operatorSelect = conditionBar.getOperatorSelect();
192 			var inputValues = conditionBar.getInputValues();
193 			var id = conditionBar.getId();
194 			var conditionPanel, negationDiv, propertyDiv, operatorDiv, valuesDiv, optionsDiv;
195 			var negationButton, helpButton, removeButton;
196 			var i;
197 
198 			queryElements.push(conditionBar);
199 			conditionPanel = createDIV("conditionPanel", "cnd" + id);
200 			negationDiv = createDIV("negation", "negationDiv" + id);
201 			propertyDiv = createDIV("property", "propertyDiv" + id);
202 			helpButton = createDIV("helpButton", "helpProperty" + id);
203 			helpTooltip = createDIV("helpTooltip", "helpTooltip" + id);
204 			operatorDiv = createDIV("operator", "operatorDiv" + id);
205 			valuesDiv = createDIV("value", "valueDiv" + id);
206 			optionsDiv = createDIV("grouping", "optionsDiv" + id);
207 			negateButton = createDIV("negationButton", "neg" + id);
208 			removeButton = createDIV("removeButton", "rem" + id);
209 
210 			for( i = 0; i < inputValues.length; ++i) {
211 				valuesDiv.append(inputValues[i]);
212 			}
213 
214 			negateButton.bind("click", function() {
215 				conditionBar.negateCondition();
216 				updateCQLCode();
217 			});
218 
219 
220 			helpButton.tooltip({
221 				position : "top right",
222 				opacity : 0.9
223 			});
224 
225 			helpTooltip.html(conditionBar.getPropertyDescription());
226 			removeButton.bind("click", removeConditionHandler);
227 
228 			negationDiv.append(negateButton);
229 			conditionPanel.append(negationDiv);
230 			propertyDiv.append(propertySelect);
231 			propertyDiv.append(helpButton);
232 			propertyDiv.append(helpTooltip);
233 			conditionPanel.append(propertyDiv);
234 			operatorDiv.append(operatorSelect);
235 			conditionPanel.append(operatorDiv);
236 			conditionPanel.append(valuesDiv);
237 			optionsDiv.append(removeButton);
238 			conditionPanel.append(optionsDiv);
239 			currentId++;
240 
241 			return conditionPanel;
242 		};

260 		/**
261 		 * Updates the values of the condition with the given array.
262 		 * @param conditionId The id of the condition to update.
263 		 * @param values The array of values to be set.
264 		 *
265 		 * @private
266 		 * @ignore
267 		 * @memberOf expcat.plugins.QueryComposerUIManager#
268 		 */
269 
270 		var updateConditionValues = function(conditionId, values) {
271 			var conditionPanel = ui.getUIComponent("cnd" + conditionId);
272 			var i;
273 
274 			$(".value", conditionPanel).empty();
275 			for( i = 0; i < values.length; ++i) {
276 				$(".value", conditionPanel).append(values[i]);
277 			}
278 
279 		}

281 		/**
282 		 * Adds a new condition (configured with default values) to the UI.
283 		 *
284 		 * @memberOf expcat.plugins.QueryComposerUIManager#
285 		 * @public
286 		 */
287 
288 		var addCondition = function() {
289 			if($("#" + uiContainerId).children().length > 0)
290 				ui.addUIComponent(createOperatorPanel());
291 			ui.addUIComponent(createConditionPanel());
292 			updateUI();
293 		};

295 		/**
296 		 * Updates and sync the UI.
297 		 *
298 		 * @private
299 		 * @ignore
300 		 * @memberOf expcat.plugins.QueryComposerUIManager#
301 		 */
302 
303 		var updateUI = function() {
304 			var root = $("#" + uiContainerId);
305 
306 			// Notice the detach to avoid the problem of losing event handlers.
307 			root.detach();
308 			root.children().detach();
309 
310 			$("#query-container").append(ui.buildDOMTreeForComponents(root));
311 			updateCQLCode();
312 		};

314 		/**
315 		 * Updates the CQL code shown in the UI.
316 		 *
317 		 * @private
318 		 * @ignore
319 		 * @memberOf expcat.plugins.QueryComposerUIManager#
320 		 */
321 
322 		var updateCQLCode = function() {
323 			var cqlCode = generateCQLCode();
324 			$("#cqlCode").html(cqlCode);
325 		}
326 
327 		/**
328 		 * Generates the CQL code for all the conditions in the UI.
329 		 *
330 		 * @return {String} The CQL code that represents all the conditions of the UI.
331 		 * @memberOf expcat.plugins.QueryComposerUIManager#
332 		 * @public
333 		 */
334 
335 		var generateCQLCode = function() {
336 			var currentElementCode;
337 			var wasPreviousNested = false;
338 			var nestingFlags = ui.getNestingFlags();
339 			var queryArray = [];
340 			var i;
341 
342 			for( i = 0; i < queryElements.length; ++i) {
343 				currentElementCode = queryElements[i].translateToCQL();
344 
345 				if(!wasPreviousNested && nestingFlags[i]) {
346 					queryArray.push("(");
347 				}
348 
349 				if(wasPreviousNested && !nestingFlags[i]) {
350 					queryArray.push(")");
351 				}
352 
353 				queryArray.push(currentElementCode);
354 				wasPreviousNested = nestingFlags[i];
355 			}
356 
357 			if(nestingFlags[i - 1])
358 				queryArray.push(")");
359 
360 			return queryArray.join(" ");
361 		}
362 
363 		/* Public API returned by the constructor */
364 		return {
365 			addCondition : addCondition,
366 			generateCQLCode : generateCQLCode
367 		};
368 	};

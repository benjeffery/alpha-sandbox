1 expcat.namespace("expcat.cql.Property");
  2 
  3 /**
  4  * @ignore
  5  */
  6 
  7 expcat.cql.Property = ( function() {
  8 
  9 	// Aliases.
 10 	var TYPES = expcat.cql.Type;
 11 	var Value = expcat.cql.Value;
 12 
 13 	/**
 14 	 * Represents the definition of an entity property. They are basically named
 15 	 * values that can have some additional attributes like maximum, minimum or a
 16 	 * list of allowed values.
 17 	 *
 18 	 * The constructor builds a property from a property definition (object).
 19 	 *
 20 	 * @param definition.name {String} The name of the property.
 21 	 * @param definition.type {String} CQL type for the property value.
 22 	 * @param definition.description {String} Brief description of the property.
 23 	 * @param definition.maximum {Number} Maximum value it can take (undefined if
 24 	 * there is no maximum).
 25 	 * @param definition.minimum {Number} Minimum value it can take (undefined if
 26 	 * there is no minimum).
 27 	 * @param definition.allowedValues {Array} Array of permitted values for the
 28 	 * property (undefined if there are no restrictions).
 29 	 *
 30 	 * @name expcat.cql.Property
 31 	 * @constructor
 32 	 * @author Jacob Almagro - ExplorerCat Project.
 33 	 */
 34 
 35 	var Property = function(definition) {
 36 
 37 		var type;
 38 		var name = definition.name || "null";
 39 		var description = definition.description || null;
 40 		var maximum = definition.maximum || null;
 41 		var minimum = definition.minimum || null;
 42 		var allowedValues = definition.allowedValues || null;
 43 
 44 		if( typeof TYPES[definition.type] === "undefined") {
 45 			throw new Error("Unknown CQL type : " + definition.type);
 46 		}
 47 		type = TYPES[definition.type];
 48 
 49 		/**
 50 		 * Gets the name of the property.
 51 		 * @return {String} The name of the property.
 52 		 *
 53 		 * @memberOf expcat.cql.Property#
 54 		 * @public
 55 		 */
 56 
 57 		var getName = function() {
 58 			return name;
 59 		}
 60 
 61 		/**
 62 		 * Gets the type of the property.
 63 		 * @return {expcat.cql.Type} The object that represents the property type.
 64 		 *
 65 		 * @memberOf expcat.cql.Property#
 66 		 * @public
 67 		 */
 68 
 69 		var getType = function() {
 70 			return type;
 71 		}
 72 
 73 		/**
 74 		 * Gets the value of the property. Since this is a property reference it will
 75 		 * return the name of the propery.
 76 		 * @return {String} The name of the property.
 77 		 *
 78 		 * @memberOf expcat.cql.Property#
 79 		 * @public
 80 		 */
 81 
 82 		var getValue = function() {
 83 			return getName();
 84 		};
 85 
 86 		/**
 87 		 * Translates the property into CQL code (this is equivalent to the name of the
 88 		 * property).
 89 		 * @return {String} The property translated into CQL code.
 90 		 *
 91 		 * @memberOf expcat.cql.Property#
 92 		 * @public
 93 		 */
 94 
 95 		var translate = function() {
 96 			return getName();
 97 		};
 98 
 99 		/**
100 		 * Gets the description of the property.
101 		 * @return {String} The property description or null.
102 		 *
103 		 * @memberOf expcat.cql.Property#
104 		 * @public
105 		 */
106 
107 		var getDescription = function() {
108 			return description;
109 		};
110 
111 		/**
112 		 * Gets the maximum value for the property.
113 		 * @retun {Number} The maximum value this property can store or null if not
114 		 * specified.
115 		 *
116 		 * @memberOf expcat.cql.Property#
117 		 * @public
118 		 */
119 
120 		var getMaximum = function() {
121 			return maximum;
122 		};
123 
124 		/**
125 		 * Gets the minimum value for the property.
126 		 * @retun {Number} The minimum value this property can store or null if not
127 		 * specified.
128 		 *
129 		 * @memberOf expcat.cql.Property#
130 		 * @public
131 		 */
132 
133 		var getMinimum = function() {
134 			return minimum;
135 		};
136 
137 		/**
138 		 * Gets an array containing the values allowed for the property.
139 		 * @return {Array} An array of strings containing the values allowed for the
140 		 * property or null if not specified.
141 		 *
142 		 * @memberOf expcat.cql.Property#
143 		 * @public
144 		 */
145 
146 		var getAllowedValues = function() {
147 			return allowedValues;
148 		};
149 
150 		/**
151 		 * Checks if the given value is valid for the property (types have to match).
152 		 * @param value {expcat.cql.Value} A data value object that contains the
153 		 * value to be checked.
154 		 * @return {Boolean} True if the value is compatible with the property, false
155 		 * otherwise.
156 		 *
157 		 * @memberOf expcat.cql.Property#
158 		 * @public
159 		 */
160 
161 		var isValueValid = function(value) {
162 
163 			if(!value instanceof Value)
164 				throw new Error("Expecting a Value object but you provided a primitive type " + typeof value);
165 
166 			// Notice that we don't allow type conversion for numeric values in this method.
167 			if(!(value.getType() === type))
168 				return false;
169 
170 			if((value.getType() === TYPES.INTEGER || value.getType() === TYPES.REAL) && isNaN(value.getValue()))
171 				return false;
172 
173 			if(maximum && value.getValue() > maximum) {
174 				return false;
175 			}
176 			if(minimum && value.getValue() < minimum) {
177 				return false;
178 			}
179 			// Linear search (small arrays anyway).
180 			if(allowedValues && allowedValues.indexOf(value.getValue()) === -1) {
181 				return false;
182 			}
183 
184 			return true;
185 		};
186 
187 		/* Public API returned by the constructor */
188 		return {
189 			getName : getName,
190 			getValue : getValue,
191 			getType : getType,
192 			translate : translate,
193 			getDescription : getDescription,
194 			getMinimum : getMinimum,
195 			getMaximum : getMaximum,
196 			getAllowedValues : getAllowedValues,
197 			isValueValid : isValueValid
198 		};
199 	};
200 
201 	/* Prototype methods and properties */
202 	Property.prototype = {
203 		constructor : Property
204 	};
205 
206 	return Property;
207 
208 }());

1 expcat.namespace("expcat.cql.Value");
  2 
  3 /**
  4  * @ignore
  5  */
  6 
  7 expcat.cql.Value = ( function() {
  8 
  9 	// Aliases.
 10 	var TYPES = expcat.cql.Type;
 11 
 12 	/**
 13 	 * Represents a CQL data value with an associated CQL type that can be translated
 14 	 * into CQL. The constructor builds a data value with the given type.
 15 	 * @see expcat.cql.Type
 16 	 * @param type {String} A string with the name of the value type.
 17 	 * @param value A representation of the value (will be cast to the given type).
 18 	 *
 19 	 * @constructor
 20 	 * @name expcat.cql.Value
 21 	 * @public
 22 	 * @author Jacob Almagro - ExplorerCat project
 23 	 */
 24 
 25 	var Value = function(type, value) {
 26 		if( typeof TYPES[type] === "undefined") {
 27 			throw new Error("Unknown CQL type : " + type);
 28 		}
 29 		type = TYPES[type];
 30 		value = type.parse(value);
 31 
 32 		/**
 33 		 * Gets the stored value.
 34 		 * @return The contained value as a primitive (cast to the assigned type).
 35 		 *
 36 		 * @memberOf expcat.cql.Value#
 37 		 * @public
 38 		 */
 39 
 40 		var getValue = function() {
 41 			return value;
 42 		};
 43 
 44 		/**
 45 		 * Gets the type of the value.
 46 		 * @return {expcat.cql.Type} The object type for this value.
 47 		 *
 48 		 * @memberOf expcat.cql.Value#
 49 		 * @public
 50 		 */
 51 
 52 		var getType = function() {
 53 			return type;
 54 		};
 55 
 56 		/**
 57 		 * Translates the value into CQL code.
 58 		 * @return {String} The equivalent CQL code for the value.
 59 		 *
 60 		 * @memberOf expcat.cql.Value#
 61 		 * @public
 62 		 */
 63 
 64 		var translate = function() {
 65 			// Date is a special case.
 66 			if(type === TYPES.DATE) {
 67 				var year = value.getFullYear();
 68 				var month = value.getMonth();
 69 				var day = value.getDate();
 70 
 71 				return year + "/" + (month < 10 ? "0" + month : month) + "/" + (day < 10 ? "0" + day : day);
 72 			}
 73 			return value;
 74 		};
 75 
 76 		/* Public API returned by the constructor. */
 77 		return {
 78 			getValue : getValue,
 79 			getType : getType,
 80 			translate : translate
 81 		};
 82 	};
 83 
 84 	/* Prototype methods. */
 85 	Value.prototype = {
 86 		constructor : Value
 87 	};
 88 
 89 	return Value;
 90 }());

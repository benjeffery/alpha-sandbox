131 	/**
132 	 * Represents a date as dd/mm/yyyy or yyyy/mm/dd where the separators can be
133 	 * slash (/) or dash (-).
134 	 * @namespace
135 	 */
136 
137 	DATE : {
138 
139 		getName : function() {
140 			return "DATE";
141 		},
142 
143 		/**
144 		 * Parses the given value as a a date. Supported formats are dd/mm/yyyy and
145 		 * yyyy/mm/dd (being the separator a slash(/) or a dash(-)).
146 		 *
147 		 * @example "2010/08/25", "25-08-2010" are valid dates whereas "2010/8/25" is
148 		 * not.
149 		 */
150 
151 		parse : function(value) {
152 			var stringValue = value.toString();
153 			var daySeparator = stringValue.charAt(2);
154 			var yearSeparator = stringValue.charAt(4);
155 			var day, month, year;
156 
157 			if((daySeparator === '-' || daySeparator === '/') && stringValue.length === 10) {
158 				day = parseInt(stringValue.substr(0, 2), 10);
159 				month = parseInt(stringValue.substr(3, 2), 10);
160 				year = parseInt(stringValue.substr(6, 4), 10);
161 			}
162 			else if((yearSeparator === '-' || yearSeparator === '/') && stringValue.length === 10) {
163 				day = parseInt(stringValue.substr(8, 2), 10);
164 				month = parseInt(stringValue.substr(5, 2), 10);
165 				year = parseInt(stringValue.substr(0, 4), 10);
166 			}
167 			else {
168 				throw new Error("Wrong date format, expecting yyyy/mm/dd or dd/mm/yyyy");
169 			}
170 
171 			// Weak validation.
172 			if(day > 31 || day < 1 || month > 12 || month < 1 || year < 0) {
173 				throw new Error("Wrong date");
174 			}
175 
176 			return new Date(year, month, day);
177 		},
178 
179 		/**
180 		 * Generates a default value for the type.
181 		 */
182 
183 		defaultValue : function() {
184 			return parse("2000-01-01");
185 		}
186

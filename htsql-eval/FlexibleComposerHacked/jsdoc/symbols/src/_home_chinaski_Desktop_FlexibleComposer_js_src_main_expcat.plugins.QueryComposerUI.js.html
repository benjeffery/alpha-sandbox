 12 expcat.plugins.QueryComposerUI = ( function() {
 13 
 14 	/**
 15 	 * Builds the user interface DOM for composing CQL queries. This implementation
 16 	 * allows only one level of nesting. The composer provides a method to build a
 17 	 * DOM tree that can be attached to the document.
 18 	 *
 19 	 * The constructor creates a new composer.
 20 	 *
 21 	 * @param nestingDivs {[jQuery]} An array containing the nesting divs (sorted
 22 	 * from outer to inner) that will be cloned when building the DOM tree for the
 23 	 * user interface.
 24 	 *
 25 	 * @name expcat.plugins.QueryComposerUI
 26 	 * @constructor
 27 	 * @author Jacob Almagro - ExplorerCat Project.
 28 	 */
 29 
 30 	var QueryComposerUI = function(nestingDivs) {
 31 
 32 		// The DOM elements that will used as nested containers.
 33 		nestingDivs = nestingDivs || [$("<div></div>")];
 34 
 35 		// Contains all the components of the UI (operator selects, conditions,etc.).
 36 		var uiComponents = [];
 37 
 38 		// Indicates the status of each component (true = nested).
 39 		var isComponentNested = [];
 40 
 41 		/**
 42 		 * Adds a new UI component to the composer.
 43 		 * @param component {jQuery} The UI component that will be
 44 		 * added to the composer.
 45 		 * @param isNested {Boolean} True if the component is nested (false by default).
 46 		 *
 47 		 * @memberOf expcat.plugins.QueryComposerUI#
 48 		 * @public
 49 		 */
 50 
 51 		var addUIComponent = function(component, isNested) {
 52 			// Notice we convert the element into a jQuery element.
 53 			uiComponents.push(component);
 54 			isComponentNested.push(isNested || false);
 55 		};
 56 
 57 		/**
 58 		 * Gets the UI component for the given id.
 59 		 * @param id {String} The identifier of the component.
 60 		 * @return {jQuery} The associated component or null if not found.
 61 		 *
 62 		 * @memberOf expcat.plugins.QueryComposerUI#
 63 		 * @public
 64 		 */
 65 
 66 		var getUIComponent = function(id) {
 67 			var index = getIndexForComponentId(id);
 68 			if(index === -1)
 69 				return null;
 70 			else
 71 				return uiComponents[index];
 72 		}
 73 
 74 		/**
 75 		 * Removes a set of consecutive components from the composer.
 76 		 * @param baseComponentId {String} The identifier of the first component to be
 77 		 * removed.
 78 		 * @param removingWindowOffset {Integer} The offset of the removing window. This
 79 		 * A value of 0 will only remove the base component, -1 will also remove the
 80 		 * previous one and so on.
 81 		 *
 82 		 * @memberOf expcat.plugins.QueryComposerUI#
 83 		 * @public
 84 		 */
 85 
 86 		var removeUIComponents = function(baseComponentId, removingWindowOffset) {
 87 			var index = getIndexForComponentId(baseComponentId);
 88 			removingWindowOffset = removingWindowOffset || 0;
 89 			var numElementsToRemove = Math.abs(removingWindowOffset) + 1;
 90 
 91 			// If the id doesn't exist we leave.
 92 			if(index === -1)
 93 				return;
 94 
 95 			if(removingWindowOffset < 0) {
 96 				index = index + removingWindowOffset;
 97 				if(index < 0)
 98 					index = 0;
 99 			}
100 
101 			uiComponents.splice(index, numElementsToRemove);
102 			isComponentNested.splice(index, numElementsToRemove);
103 		};
104 
105 		/**
106 		 * Gets the number of components registered in the composer.
107 		 * @return {Integer} Number of registered components.
108 		 *
109 		 * @memberOf expcat.plugins.QueryComposerUI#
110 		 * @public
111 		 */
112 
113 		var getNumUIComponents = function() {
114 			return uiComponents.length;
115 		}
116 
117 		/**
118 		 * Private method that translates a component id into an array index.
119 		 * @return {Integer} The index for the given id or -1 if not found.
120 		 *
121 		 * @private
122 		 * @ignore
123 		 * @memberOf expcat.plugins.QueryComposerUI#
124 		 */
125 
126 		var getIndexForComponentId = function(id) {
127 			var i;
128 			for( i = 0; i < uiComponents.length; ++i) {
129 				if(uiComponents[i].attr("id") === id)
130 					return i;
131 			}
132 			return -1;
133 		};
134 
135 		/**
136 		 * Sets the nesting level (true/false) of the given component.
137 		 * @param componentId {String} The identifier of the component.
138 		 * @param isNested {Boolean} True if the component is nested, false otherwise.
139 		 *
140 		 * @memberOf expcat.plugins.QueryComposerUI#
141 		 * @public
142 		 */
143 
144 		var setComponentNesting = function(componentId, isNested) {
145 			var index = getIndexForComponentId(componentId);
146 			isComponentNested[index] = isNested;
147 		};
148 
149 		/**
150 		 * Sets the nesting level (true/false) of the given component and all the
151 		 * component within the given radius around it.
152 		 * @param componentId {String} The identifier of the component.
153 		 * @param radius {Integer} Number of components above/below that will be
154 		 * affected.
155 		 * @param areNested {Boolean} True if the components are nested, false otherwise.
156 		 *
157 		 * @memberOf expcat.plugins.QueryComposerUI#
158 		 * @public
159 		 */
160 
161 		var setComponentNestingWithinRadius = function(componentId, radius, areNested) {
162 			var index = getIndexForComponentId(componentId);
163 			var i;
164 
165 			isComponentNested[index] = areNested;
166 			i = index - radius;
167 			if(i >= 0) {
168 				while(i < index) {
169 					isComponentNested[i] = areNested;
170 					i++;
171 				}
172 			}
173 			i = index + radius;
174 			if(i < isComponentNested.length) {
175 				while(i > index) {
176 					isComponentNested[i] = areNested;
177 					i--;
178 				}
179 			}
180 		};
181 
182 		/**
183 		 * Applies the given function to the array of nesting flags.
184 		 * The function is in charge of correcting the values of this array.
185 		 * @param correctionFunction {Function} Function that will correct any incoherent
186 		 * nesting.
187 		 *
188 		 * @memberOf expcat.plugins.QueryComposerUI#
189 		 * @public
190 		 */
191 
192 		var applyNestingCorrectorFunction = function(correctorFunction) {
193 			correctorFunction(isComponentNested);
194 		}
195 
196 		/**
197 		 * Checks if the given component is nested.
198 		 * @param id {String} Identifier of the component.
199 		 * @return True if the component is nested.
200 		 *
201 		 * @memberOf expcat.plugins.QueryComposerUI#
202 		 * @public
203 		 */
204 
205 		var checkComponentIsNested = function(id) {
206 			var index = getIndexForComponentId(id);
207 			if(index === -1) {
208 				return false;
209 			}
210 			else {
211 				return isComponentNested[i];
212 			}
213 		}
214 
215 		/**
216 		 * Checks if the component that precedes the given one is nested.
217 		 * @param id {String} Identifier of the component.
218 		 * @return {Boolean} True if the preceding component is nested.
219 		 *
220 		 * @memberOf expcat.plugins.QueryComposerUI#
221 		 * @public
222 		 */
223 
224 		var checkPreviousComponentIsNested = function(id) {
225 			var index = getIndexForComponentId(id);
226 			if(index === -1 || index === 0) {
227 				return false;
228 			}
229 			else {
230 				return isComponentNested[index - 1];
231 			}
232 		}
233 
234 		/**
235 		 * Checks if the component that follows the given one is nested.
236 		 * @param id {String} Identifier of the component.
237 		 * @return {Boolean} True if the following component is nested.
238 		 *
239 		 * @memberOf expcat.plugins.QueryComposerUI#
240 		 * @public
241 		 */
242 
243 		var checkNextComponentIsNested = function(id) {
244 			var index = getIndexForComponentId(id);
245 			if(index === -1 || index === isComponentNested.length) {
246 				return false;
247 			}
248 			else {
249 				return isComponentNested[index + 1];
250 			}
251 		}
252 
253 		/**
254 		 * Gets the number of consecutive nested components, starting to count from the
255 		 * given one.
256 		 * @param id {String} Identifier of the component.
257 		 * @return {Integer} The number of consecutive nested components that follow the
258 		 * given one.
259 		 *
260 		 * @memberOf expcat.plugins.QueryComposerUI#
261 		 * @public
262 		 */
263 
264 		var getNumConsecutiveNestedComponents = function(id) {
265 			var index = getIndexForComponentId(id);
266 			var i;
267 
268 			if(index === -1) {
269 				return 0;
270 			}
271 
272 			for( i = index; i < isComponentNested.length; ++i) {
273 				if(!isComponentNested[i])
274 					return i - index;
275 			}
276 
277 			return i - index - 1;
278 		}
279 
280 		/**
281 		 * Builds a DOM tree, attaching all the components registered in the composer
282 		 * (with the proper configuration of nested containers) to the given root.
283 		 * @param root {jQuery} The element that will be used as the root of the tree.
284 		 * @return {jQuery} The root element of the DOM tree, ready to be
285 		 * attached to the document.
286 		 *
287 		 * @memberOf expcat.plugins.QueryComposerUI#
288 		 * @public
289 		 */
290 
291 		var buildDOMTreeForComponents = function(root) {
292 			root = root || "<div></div>";
293 			var wasPreviousNested = false;
294 			var currentContainer = root;
295 			root = $(root);
296 			var i;
297 
298 			for( i = 0; i < uiComponents.length; ++i) {
299 				if(!wasPreviousNested && isComponentNested[i]) {
300 					currentContainer = createNestingContainer();
301 					root.append(currentContainer);
302 					currentContainer = getInnerChild(currentContainer);
303 					currentContainer.append(uiComponents[i]);
304 				}
305 				else if(wasPreviousNested && !isComponentNested[i]) {
306 					currentContainer = root;
307 					currentContainer.append(uiComponents[i]);
308 				}
309 				else {
310 					currentContainer.append(uiComponents[i]);
311 				}
312 				wasPreviousNested = isComponentNested[i];
313 			}
314 
315 			return root;
316 		};
317 
318 		/**
319 		 * Private method that creates the nesting container where nested components will
320 		 * be appended.
321 		 * @return {jQuery} The root of the DOM subtree that represent the
322 		 * containers.
323 		 *
324 		 * @private
325 		 * @ignore
326 		 * @memberOf expcat.plugins.QueryComposerUI#
327 		 */
328 
329 		var createNestingContainer = function() {
330 			var i;
331 			var childContainer = null;
332 			var root = $(nestingDivs[0]).clone();
333 			var nestingContainer = root;
334 
335 			for( i = 1; i < nestingDivs.length; ++i) {
336 				childContainer = $(nestingDivs[i]).clone();
337 				nestingContainer.append(childContainer);
338 				nestingContainer = childContainer;
339 			}
340 
341 			return root;
342 		};
343 
344 		/**
345 		 * Gets an array of boolean flags that indicates if a component is nested (true)
346 		 * or not (false).
347 		 * @return {[Boolean]} An array of booleans that specifies which components are
348 		 * nested (true)
349 		 */
350 
351 		var getNestingFlags = function() {
352 			return isComponentNested;
353 		}
354 
355 		/**
356 		 * Gets the deepest child of the given root. In case of multiple children, only
357 		 * the first one is considered.
358 		 * @return {jQuery} The deepest child of the given element.
359 		 * @private
360 		 * @ignore
361 		 * @memberOf expcat.plugins.QueryComposerUI#
362 		 */
363 
364 		var getInnerChild = function(root) {
365 			var children = root.children();
366 			while(children.length !== 0) {
367 				root = $(children[0]);
368 				children = root.children();
369 			}
370 
371 			return root;
372 		};
373 
374 		/* Public API returned by the constructor */
375 		return {
376 			addUIComponent : addUIComponent,
377 			removeUIComponents : removeUIComponents,
378 			getUIComponent : getUIComponent,
379 			getNumUIComponents : getNumUIComponents,
380 			setComponentNesting : setComponentNesting,
381 			setComponentNestingWithinRadius : setComponentNestingWithinRadius,
382 			applyNestingCorrectorFunction : applyNestingCorrectorFunction,
383 			checkComponentIsNested : checkComponentIsNested,
384 			checkPreviousComponentIsNested : checkPreviousComponentIsNested,
385 			checkNextComponentIsNested : checkNextComponentIsNested,
386 			getNumConsecutiveNestedComponents : getNumConsecutiveNestedComponents,
387 			getNestingFlags : getNestingFlags,
388 			buildDOMTreeForComponents : buildDOMTreeForComponents,
389 		};
390 	};
391 
392 	/* Prototype methods and properties */
393 	QueryComposerUI.prototype = {
394 		constructor : QueryComposerUI
395 	};
396 
397 	return QueryComposerUI;
398 
399 }());
